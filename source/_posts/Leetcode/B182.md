## 3921 - Score Validator

**题意**
题目链接：[score-validator](https://leetcode.com/problems/score-validator/description/)

给定字符串数组 `events`，初始 `score = 0，counter = 0`。从左到右处理：
- 若为数字 `"0","1","2","3","4","6"`，加到 `score`
- 若为 `"W"`，`counter += 1`
- 若为 `"WD"` 或 `"NB"`，`score += 1`

当 `counter == 10` 或处理完毕时停止，返回 `[score, counter]`。

**解题思路**

直接模拟遍历，按规则更新即可。

**代码**

```cpp
class Solution {
public:
    // scoreValidator - 计算最终得分和计数器值
    vector<int> scoreValidator(vector<string>& events) {
        int sco = 0;
        int cnt = 0;

        for (string& ev : events) {
            if (cnt == 10) break;

            if (ev == "W") {
                cnt++;
            } else if (ev == "WD" || ev == "NB") {
                sco++;
            } else {
                sco += stoi(ev);
            }
        }

        return {sco, cnt};
    }
};
```

**复杂度分析**

时间复杂度：$O(n)$，最多遍历一次数组。

空间复杂度：$O(1)$，只使用常数个变量。

## 3922 - Minimum Flips to Make Binary String Coherent

**题意**
题目链接：[minimum-flips-to-make-binary-string-coherent](https://leetcode.com/problems/minimum-flips-to-make-binary-string-coherent/description/)

给定二进制字符串 `s`，一次操作可翻转任意字符。若字符串不包含子序列 `"011"` 或 `"110"`，则称其为 coherent。求最少翻转次数使 `s` 变成 coherent。

**解题思路**

先分析合法字符串的结构。对于每个 `'0'`，其前后最多只能各有一个 `'1'`，否则会形成 forbidden subsequence。

合法字符串只有三类：
1. 最多只有一个 `'1'`：代价为 $\max(0, cnt_1 - 1)$
2. 全为 `'1'`：代价为 $cnt_0$
3. 恰好两个 `'1'`，且必须在首尾：代价为 `(s[0]=='0') + (s[n-1]=='0') + 中间'1'的数量`

另有一种更通用的 **自动机 DP** 解法：用 4 位状态记录是否出现过子序列前缀 `"0"`, `"1"`, `"01"`, `"11"`，转移时排除会形成 `"011"` 或 `"110"` 的情况。

**代码（分类讨论）**

```cpp
class Solution {
public:
    // minFlips - 计算使字符串coherent的最少翻转次数
    int minFlips(string s) {
        int n = s.size();
        int c1 = 0;

        for (char ch : s) {
            if (ch == '1') c1++;
        }

        int c0 = n - c1;

        int ans = min(max(0, c1 - 1), c0);

        if (n >= 2) {
            int cur = 0;

            if (s[0] == '0') cur++;
            if (s[n - 1] == '0') cur++;

            for (int i = 1; i + 1 < n; i++) {
                if (s[i] == '1') cur++;
            }

            ans = min(ans, cur);
        }

        return ans;
    }
};
```

**代码（自动机 DP）**

```cpp
class Solution {
public:
    // go - 计算追加字符后的新状态，非法返回-1
    int go(int ms, char ch) {
        int a0 = ms & 1;
        int a1 = ms & 2;
        int a01 = ms & 4;
        int a11 = ms & 8;

        if (ch == '0') {
            if (a11) return -1;
            ms |= 1;
        } else {
            if (a01) return -1;
            if (a0) ms |= 4;
            if (a1) ms |= 8;
            ms |= 2;
        }

        return ms;
    }

    // minFlips - 自动机DP求最少翻转次数
    int minFlips(string s) {
        const int INF = 1e9;
        vector<int> dp(16, INF);
        dp[0] = 0;

        for (char ch : s) {
            vector<int> ndp(16, INF);

            for (int ms = 0; ms < 16; ms++) {
                if (dp[ms] == INF) continue;

                for (char nx : {'0', '1'}) {
                    int ns = go(ms, nx);
                    if (ns == -1) continue;

                    int add = ch != nx;
                    ndp[ns] = min(ndp[ns], dp[ms] + add);
                }
            }

            dp = ndp;
        }

        return *min_element(dp.begin(), dp.end());
    }
};
```

**复杂度分析**

时间复杂度：$O(n)$，空间复杂度：$O(1)$。

## 3923 - Minimum Generations to Target Point

**题意**
题目链接：[minimum-generations-to-target-point](https://leetcode.com/problems/minimum-generations-to-target-point/description/)

给定三维点集 `points`，第 $k$ 代由第 $0$ 到 $k-1$ 代中所有点两两配对生成中点 `floor((a+b)/2)`。求目标点 `target` 最早出现在哪一代，若无法生成返回 $-1$。

**解题思路**

坐标范围为 $0..6$，总状态数仅 $7^3 = 343$，可直接分层模拟。维护已出现点集合 `all`，每代只枚举当前已有点生成新点，新点加入下一代。

**代码**

```cpp
class Solution {
public:
    // enc - 将三维点编码为整数
    int enc(const vector<int>& p) {
        const int B = 7;
        return p[0] * B * B + p[1] * B + p[2];
    }

    // minGenerations - 计算目标点最早出现的代数
    int minGenerations(vector<vector<int>>& points, vector<int>& target) {
        const int B = 7;
        const int TOT = 343;
        int tar = enc(target);

        vector<int> vis(TOT, 0);
        vector<int> all;

        for (auto& p : points) {
            int id = enc(p);
            vis[id] = 1;
            all.push_back(id);
        }

        if (vis[tar]) return 0;

        for (int ans = 1; ; ans++) {
            vector<int> add;
            int m = all.size();

            for (int i = 0; i < m; i++) {
                int a = all[i];
                int ax = a / (B * B);
                int ay = a / B % B;
                int az = a % B;

                for (int j = i + 1; j < m; j++) {
                    int b = all[j];
                    int bx = b / (B * B);
                    int by = b / B % B;
                    int bz = b % B;

                    int cx = (ax + bx) / 2;
                    int cy = (ay + by) / 2;
                    int cz = (az + bz) / 2;

                    int c = cx * B * B + cy * B + cz;

                    if (c == tar) return ans;

                    if (!vis[c]) {
                        vis[c] = 1;
                        add.push_back(c);
                    }
                }
            }

            if (add.empty()) return -1;

            for (int id : add) {
                all.push_back(id);
            }
        }
    }
};
```

**复杂度分析**

状态空间 $U = 343$ 为常数，每代枚举 $O(U^2)$ 点对，最多 $U$ 代。

时间复杂度：$O(U^3)$，空间复杂度：$O(U)$。

## 3924 - Minimum Threshold Path With Limited Heavy Edges

**题意**
题目链接：[minimum-threshold-path-with-limited-heavy-edges](https://leetcode.com/problems/minimum-threshold-path-with-limited-heavy-edges/description/)

给定无向带权图，阈值决定边为轻 (权值 $\le$ 阈值) 或重 (权值 $>$ 阈值)。求最小阈值使得从 `source` 到 `target` 存在一条至多含 $k$ 条重边的路径。

**解题思路**

阈值具有单调性：更大的阈值只会让重边变少，不会增多。可二分答案。

判定：给定阈值后，将每条边代价设为 $0$ (轻) 或 $1$ (重)，用 **0-1 BFS** 求从 `source` 到 `target` 的最少重边数，若 $\le k$ 则可行。

**代码**

```cpp
class Solution {
public:
    // ok - 判断阈值lim下是否存在重边数不超过k的路径
    bool ok(int n, vector<vector<pair<int, int>>>& g, int s, int t, int k, int lim) {
        const int INF = 1e9;
        vector<int> dis(n, INF);
        deque<int> dq;

        dis[s] = 0;
        dq.push_back(s);

        while (!dq.empty()) {
            int u = dq.front();
            dq.pop_front();

            for (auto [v, w] : g[u]) {
                int c = w > lim ? 1 : 0;

                if (dis[v] > dis[u] + c) {
                    dis[v] = dis[u] + c;

                    if (c == 0) {
                        dq.push_front(v);
                    } else {
                        dq.push_back(v);
                    }
                }
            }
        }

        return dis[t] <= k;
    }

    // minimumThreshold - 求最小阈值使存在至多k条重边的路径
    int minimumThreshold(int n, vector<vector<int>>& edges, int source, int target, int k) {
        if (source == target) return 0;

        vector<vector<pair<int, int>>> g(n);
        int mx = 0;

        for (auto& e : edges) {
            int u = e[0];
            int v = e[1];
            int w = e[2];

            g[u].push_back({v, w});
            g[v].push_back({u, w});
            mx = max(mx, w);
        }

        int l = 0;
        int r = mx;
        int ans = -1;

        while (l <= r) {
            int mid = l + (r - l) / 2;

            if (ok(n, g, source, target, k, mid)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }

        return ans;
    }
};
```

**复杂度分析**

每次判定用 0-1 BFS，时间 $O(n + m)$，二分 $O(\log W)$ 次。

总时间复杂度：$O((n + m) \log W)$，空间复杂度：$O(n + m)$。
